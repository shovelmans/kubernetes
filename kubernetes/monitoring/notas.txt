a la hora de crear la app en argo hacer como yaml y poner
apiVersion: argoproj.io/v1alpha1
kind: Application
metadata:
  name: monitoring
  namespace: argocdinicio           # <- donde está ArgoCD
spec:
  project: default
  destination:
    server: https://kubernetes.default.svc
    namespace: monitoring           # <- donde se desplegarán los recursos
  source:
    repoURL: https://github.com/shovelmans/kubernetes.git
    targetRevision: main
    path: kubernetes/monitoring
    helm:
      skipCrds: true                # <- CLAVE: ArgoCD invoca Helm con --skip-crds
  syncPolicy:
    syncOptions:
      - CreateNamespace=true
